import numpy as np
import time
import matplotlib.pyplot as plt
def bubble_sort(a):#
    for i in range(1, len(a)):
        for j in range(i, 0, -1):
            if a[j] >= a[j-1]:
                break
            else:
                a[j], a[j-1] = a[j-1], a[j]
    return a
def heap_sort(array):#
    i = 0
    n = len(array)

    while(i < n):
        # 
        upheap(array, i)
        i += 1
    while(i > 1):
        # 
        i -= 1
        tmp = array[0]
        array[0] = array[i]
        array[i] = tmp
        # 
        downheap(array, i-1)
    return array
def upheap(array, n):# array[n](0n-1)
    while n != 0:
        parent = int((n - 1) / 2)
        if array[n] > array[parent]:
            # 
            tmp = array[n]
            array[n] = array[parent]
            array[parent] = tmp
            n = parent
        else:
            break
def downheap(array, n):# [0](0n)
    if n == 0: return
    parent = 0
    while True:
        child = 2 * parent + 1 # array[n]
        if child > n: break
        if (child < n) and array[child] < array[child + 1]:
            child += 1
        if array[parent] < array[child]: # 
            tmp = array[child]
            array[child] = array[parent]
            array[parent] = tmp
            parent = child; # 
        else:
            break
